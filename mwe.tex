\usepackage[]{defield}


\begin{document}

% minimal
%
\begin{tikzpicture} [
declare function={f(\x,\y)=(1-\x)*\y-\x^2;},
]
\defield[
  no axes,
  xmin=-3, xmax=3,
  ymin=-4, ymax=2,
]
\end{tikzpicture}

% typical use
%
\begin{tikzpicture} [
declare function={f(\x,\y)=(1-\x)*\y-\x^2;},
]
\defield[
  title={\( y'=(1-x)y-x^2\)},
  x_0=0, y_0=1,
  xmin=-3, xmax=3,
  ymin=-4, ymax=2,
]
\end{tikzpicture}


% styled
%
\begin{tikzpicture} [
declare function={g(\x,\y)=(1-\x)*\y^2-\x^2;}
]
\defield[
  title={\( y'=(1-x)y^2-x^2\)},
  function=g,
  x_0=0, y_0=1,
  xmin=-3, xmax=4,
  ymin=-4, ymax=3,
  x samples=48,% slope columns
  y samples=24,% slope rows
  slope style={orange, line width=0.75pt},
  curve style={violet, latex-latex, line width=3pt},
  no axes,
][%  inserted after plot
%\tkzAxeXY
\tkzGrid
][% inserted before plot
%
]


\end{tikzpicture}
